import CustomItemTooltip from '@site/src/components/CustomItemTooltip';
import Item, { calculateItemStats } from '@site/src/components/Item';
import basesData from '@site/static/data/items/bases.json';
import prefixesData from '@site/static/data/items/prefixes.json';
import suffixesData from '@site/static/data/items/suffixes.json';

export const TEST_ITEM = "Spear"; // Change this to test different weapons
export const TEST_PREFIX = "Quintas"; // Change this to test different prefixes
export const TEST_SUFFIX = "of the Chasm"; // Change this to test different suffixes

## Weapon Damage Calculations

export const WeaponStats = () => {
  const baseItem = basesData.find(item => item.name === TEST_ITEM);
  const prefix = prefixesData.find(p => p.name === TEST_PREFIX);
  const suffix = suffixesData.find(s => s.name === TEST_SUFFIX);
  
  const green = calculateItemStats(baseItem, 'green', false, prefix, suffix);
  const greenPlus = calculateItemStats(baseItem, 'green', true, prefix, suffix);
  const blue = calculateItemStats(baseItem, 'blue', false, prefix, suffix);
  const bluePlus = calculateItemStats(baseItem, 'blue', true, prefix, suffix);
  const purple = calculateItemStats(baseItem, 'purple', false, prefix, suffix);
  const purplePlus = calculateItemStats(baseItem, 'purple', true, prefix, suffix);
  const orange = calculateItemStats(baseItem, 'orange', false, prefix, suffix);
  const orangePlus = calculateItemStats(baseItem, 'orange', true, prefix, suffix);
  const red = calculateItemStats(baseItem, 'red', false, prefix, suffix);
  const redPlus = calculateItemStats(baseItem, 'red', true, prefix, suffix);
  
  return (
    <div>
      <div><strong>{TEST_PREFIX} {TEST_ITEM} {TEST_SUFFIX}</strong></div>
      <div>Green: Damage:  {green.damage?.min} - {green.damage?.max}</div>
      <div>Green+: Damage:  {greenPlus.damage?.min} - {greenPlus.damage?.max}</div>
      <div>Blue: Damage:  {blue.damage?.min} - {blue.damage?.max}</div>
      <div>Blue+: Damage:  {bluePlus.damage?.min} - {bluePlus.damage?.max}</div>
      <div>Purple: Damage:  {purple.damage?.min} - {purple.damage?.max}</div>
      <div>Purple+: Damage:  {purplePlus.damage?.min} - {purplePlus.damage?.max}</div>
      <div>Orange: Damage:  {orange.damage?.min} - {orange.damage?.max}</div>
      <div>Orange+: Damage:  {orangePlus.damage?.min} - {orangePlus.damage?.max}</div>
      <div>Red: Damage:  {red.damage?.min} - {red.damage?.max}</div>
      <div>Red+: Damage:  {redPlus.damage?.min} - {redPlus.damage?.max}</div>
    </div>
  );
};

<WeaponStats />
