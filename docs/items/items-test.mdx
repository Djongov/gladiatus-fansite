import CustomItemTooltip from '@site/src/components/CustomItemTooltip';
import Item, { calculateItemStats } from '@site/src/components/Item';
import basesData from '@site/static/data/items/bases.json';

export const TEST_ITEM = "Gut graze"; // Change this to test different weapons

## Weapon Damage Calculations

export const WeaponStats = () => {
  const baseItem = basesData.find(item => item.name === TEST_ITEM);
  
  const green = calculateItemStats(baseItem, 'green', false);
  const greenPlus = calculateItemStats(baseItem, 'green', true);
  const blue = calculateItemStats(baseItem, 'blue', false);
  const bluePlus = calculateItemStats(baseItem, 'blue', true);
  const purple = calculateItemStats(baseItem, 'purple', false);
  const purplePlus = calculateItemStats(baseItem, 'purple', true);
  const orange = calculateItemStats(baseItem, 'orange', false);
  const orangePlus = calculateItemStats(baseItem, 'orange', true);
  const red = calculateItemStats(baseItem, 'red', false);
  const redPlus = calculateItemStats(baseItem, 'red', true);
  
  return (
    <div>
      <div><strong>{TEST_ITEM}</strong></div>
      <div>Green: Dur {green.durability} / Cond {green.conditioning.max}</div>
      <div>Green+: Dur {greenPlus.durability} / Cond {greenPlus.conditioning.max}</div>
      <div>Blue: Dur {blue.durability} / Cond {blue.conditioning.max}</div>
      <div>Blue+: Dur {bluePlus.durability} / Cond {bluePlus.conditioning.max}</div>
      <div>Purple: Dur {purple.durability} / Cond {purple.conditioning.max}</div>
      <div>Purple+: Dur {purplePlus.durability} / Cond {purplePlus.conditioning.max}</div>
      <div>Orange: Dur {orange.durability} / Cond {orange.conditioning.max}</div>
      <div>Orange+: Dur {orangePlus.durability} / Cond {orangePlus.conditioning.max}</div>
      <div>Red: Dur {red.durability} / Cond {red.conditioning.max}</div>
      <div>Red+: Dur {redPlus.durability} / Cond {redPlus.conditioning.max}</div>
    </div>
  );
};

<WeaponStats />
